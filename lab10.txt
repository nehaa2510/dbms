3.CREATE OR REPLACE TRIGGER trg_check_instructor
BEFORE INSERT OR UPDATE ON instructor
FOR EACH ROW
DECLARE
    b department.budget%TYPE;
    i NUMBER;
BEGIN
    -- Name check
    FOR i IN 1..LENGTH(:NEW.name) LOOP
        IF NOT (SUBSTR(:NEW.name,i,1) BETWEEN 'A' AND 'Z' OR
                SUBSTR(:NEW.name,i,1) BETWEEN 'a' AND 'z') THEN
            RAISE_APPLICATION_ERROR(-20001,'Invalid Name');
        END IF;
    END LOOP;

    -- Salary check
    IF :NEW.salary <= 0 THEN
        RAISE_APPLICATION_ERROR(-20002,'Invalid Salary');
    END IF;

    -- Budget check
    SELECT budget INTO b FROM department
    WHERE dept_name = :NEW.dept_name;

    IF :NEW.salary > b THEN
        RAISE_APPLICATION_ERROR(-20003,'Exceeds Budget');
    END IF;
END;

--  Account Table

    employee_id NUMBER PRIMARY KEY,
    employee_name VARCHAR2(100) NOT NULL,

    branch_name VARCHAR2(50),

    FOREIGN KEY (branch_name) REFERENCES branch(branch_name)

--  Depositor Table

CREATE OR REPLACE TRIGGER trg_check_transaction
BEFORE INSERT OR UPDATE ON account
FOR EACH ROW
DECLARE
    v_balance account.balance%TYPE;
BEGIN
    -- Check if the account number exists (only relevant for UPDATE)
    BEGIN
        SELECT balance INTO v_balance 
        FROM account 
        WHERE account_number = :NEW.account_number;
    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            RAISE_APPLICATION_ERROR(-20001, 'Invalid account number.');
    END;

    -- Check that transaction amount is positive and non-zero
    IF :NEW.balance <= 0 THEN
        RAISE_APPLICATION_ERROR(-20002, 'Transaction amount must be positive and non-zero.');
    END IF;

    -- Check withdrawal: ensure new balance does not go below zero
    IF :NEW.balance < 0 THEN
        IF ABS(:NEW.balance) > v_balance THEN
            RAISE_APPLICATION_ERROR(-20003, 'Withdrawal exceeds available balance.');
        END IF;
    END IF;

END;
/
Trigger Created.
