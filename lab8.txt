1. DECLARE
   -- Cursor to select instructors of a given department
   CURSOR c_instructors IS
      SELECT Instructor_Id, Salary
      FROM Instructor
      WHERE Dept_No = &dept_no;  -- department number provided at runtime

   -- Variables to hold cursor values
   v_instructor_id Instructor.Instructor_Id%TYPE;
   v_salary        Instructor.Salary%TYPE;
   v_raise_amt     NUMBER;
BEGIN
   -- Open cursor
   OPEN c_instructors;

   LOOP
      -- Fetch each instructor record
      FETCH c_instructors INTO v_instructor_id, v_salary;

      -- Exit when no more rows
      EXIT WHEN c_instructors%NOTFOUND;

      -- Calculate raise amount (5% of current salary)
      v_raise_amt := v_salary * 0.05;

      -- Update salary in Instructor table
      UPDATE Instructor
      SET Salary = Salary + v_raise_amt
      WHERE Instructor_Id = v_instructor_id;

      -- Insert record into salary_raise table
      INSERT INTO salary_raise (Instructor_Id, Raise_date, Raise_amt)
      VALUES (v_instructor_id, SYSDATE, v_raise_amt);
   END LOOP;

   -- Close cursor
   CLOSE c_instructors;

   -- Commit changes
   COMMIT;
END;
/

DECLARE
   CURSOR c_course IS
      SELECT c.course_id,
             c.title,
             c.dept_name,
             c.credits,
             i.name AS instructor_name,
             s.building,
             s.room_number,
             s.time_slot_id,
             COUNT(t.ID) AS tot_student_no
      FROM course c
      JOIN section s
        ON c.course_id = s.course_id
      JOIN teaches te
        ON s.course_id = te.course_id
       AND s.sec_id    = te.sec_id
       AND s.semester  = te.semester
       AND s.year      = te.year
      JOIN instructor i
        ON te.ID = i.ID
      LEFT JOIN takes t
        ON s.course_id = t.course_id
       AND s.sec_id    = t.sec_id
       AND s.semester  = t.semester
       AND s.year      = t.year
      GROUP BY c.course_id, c.title, c.dept_name, c.credits,
               i.name, s.building, s.room_number, s.time_slot_id
      ORDER BY c.course_id;
BEGIN
   -- Cursor FOR loop automatically opens, fetches, and closes
   FOR rec IN c_course LOOP
      DBMS_OUTPUT.PUT_LINE('Course ID: ' || rec.course_id ||
                           ', Title: ' || rec.title ||
                           ', Dept: ' || rec.dept_name ||
                           ', Credits: ' || rec.credits ||
                           ', Instructor: ' || rec.instructor_name ||
                           ', Building: ' || rec.building ||
                           ', Room: ' || rec.room_number ||
                           ', Time Slot: ' || rec.time_slot_id ||
                           ', Total Students: ' || rec.tot_student_no);
   END LOOP;
END;

DECLARE
   CURSOR c_students IS
      SELECT s.ID, s.name, s.dept_name, s.tot_cred,
             t.course_id, t.sec_id, t.semester, t.year
      FROM student s
      JOIN takes t
        ON s.ID = t.ID
      WHERE t.course_id = 'CS101';

BEGIN
   -- Cursor FOR loop automatically handles OPEN/FETCH/CLOSE
   FOR rec IN c_students LOOP
      IF rec.tot_cred < 30 THEN
         -- Deregister student from CS101
         DELETE FROM takes
         WHERE ID = rec.ID
           AND course_id = rec.course_id
           AND sec_id = rec.sec_id
           AND semester = rec.semester
           AND year = rec.year;

         DBMS_OUTPUT.PUT_LINE('Student ' || rec.ID || ' (' || rec.name ||
                              ') deregistered from course ' || rec.course_id ||
                              ' due to tot_cred < 30.');
      ELSE
         DBMS_OUTPUT.PUT_LINE('Student ' || rec.ID || ' (' || rec.name ||
                              ') remains registered in ' || rec.course_id ||
                              ' with tot_cred = ' || rec.tot_cred);
      END IF;
   END LOOP;

   COMMIT;
END;

DECLARE
   CURSOR c_students IS
      SELECT ID, tot_cred, LetterGrade
      FROM student
      FOR UPDATE;  -- allows WHERE CURRENT OF

   v_id        student.ID%TYPE;
   v_tot_cred  student.tot_cred%TYPE;
   v_grade     student.LetterGrade%TYPE;
BEGIN
   FOR rec IN c_students LOOP
      -- Map GPA/credits to letter grade
      IF rec.tot_cred >= 90 THEN
         v_grade := 'A';
      ELSIF rec.tot_cred >= 75 THEN
         v_grade := 'B';
      ELSIF rec.tot_cred >= 60 THEN
         v_grade := 'C';
      ELSIF rec.tot_cred >= 45 THEN
         v_grade := 'D';
      ELSE
         v_grade := 'F';
      END IF;

      -- Update current row using WHERE CURRENT OF
      UPDATE student
      SET LetterGrade = v_grade
      WHERE CURRENT OF c_students;

      DBMS_OUTPUT.PUT_LINE('Student ' || rec.ID ||
                           ' updated to grade ' || v_grade);
   END LOOP;

   COMMIT;
END;

DECLARE
   -- Parameterized cursor: takes advisor ID as input
   CURSOR c_students (p_instructor_id VARCHAR2) IS
      SELECT s.ID, s.name, s.dept_name, s.tot_cred,
             t.course_id, t.sec_id, t.semester, t.year
      FROM student s
      JOIN advisor a
        ON s.ID = a.s_ID
      JOIN takes t
        ON s.ID = t.ID
      JOIN teaches te
        ON t.course_id = te.course_id
       AND t.sec_id    = te.sec_id
       AND t.semester  = te.semester
       AND t.year      = te.year
      WHERE a.i_ID = p_instructor_id
        AND te.ID = a.i_ID;  -- advisor is the one teaching

BEGIN
   -- Prompt user for advisor ID
   FOR rec IN c_students('&advisor_id') LOOP
      DBMS_OUTPUT.PUT_LINE('Student ID: ' || rec.ID ||
                           ', Name: ' || rec.name ||
                           ', Dept: ' || rec.dept_name ||
                           ', Credits: ' || rec.tot_cred ||
                           ', Registered Course: ' || rec.course_id ||
                           ', Section: ' || rec.sec_id ||
                           ', Semester: ' || rec.semester ||
                           ', Year: ' || rec.year);
   END LOOP;
END;
