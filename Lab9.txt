  2  BEGIN

  4      FROM (
  5          SELECT C.title, COUNT(T.ID)
  6          FROM Course C JOIN Takes T ON C.course_id = T.course_id
  7          WHERE C.dept_name = p_dept

  9          ORDER BY COUNT(T.ID) DESC

SQL> CREATE OR REPLACE PROCEDURE calc_interest(
  2      p_principle IN NUMBER,
  3      p_rate IN NUMBER,
  4      p_years IN NUMBER,
  5      p_si OUT NUMBER,
  6      p_ci OUT NUMBER,
  7      p_total IN OUT NUMBER -- Using this to return (Principle + SI)
  8  ) AS
  9  BEGIN
 10      p_si := (p_principle * p_rate * p_years) / 100;
 11      p_ci := p_principle * POWER((1 + p_rate/100), p_years) - p_principle;
 12      p_total := p_principle + p_si;
 13  END;
 14  /
