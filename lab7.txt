2. SET SERVEROUTPUT ON;

DECLARE
   v_rollno   StudentTable.RollNo%TYPE;
   v_gpa      StudentTable.GPA%TYPE;
   v_grade    VARCHAR2(2);
BEGIN
   -- Prompt user for RollNo
   v_rollno := &RollNo;

   -- Fetch GPA of the given student
   SELECT GPA
   INTO v_gpa
   FROM StudentTable
   WHERE RollNo = v_rollno;

   -- Determine letter grade
   IF v_gpa BETWEEN 0 AND 4 THEN
      v_grade := 'F';
   ELSIF v_gpa BETWEEN 4 AND 5 THEN
      v_grade := 'E';
   ELSIF v_gpa BETWEEN 5 AND 6 THEN
      v_grade := 'D';
   ELSIF v_gpa BETWEEN 6 AND 7 THEN
      v_grade := 'C';
   ELSIF v_gpa BETWEEN 7 AND 8 THEN
      v_grade := 'B';
   ELSIF v_gpa BETWEEN 8 AND 9 THEN
      v_grade := 'A';
   ELSIF v_gpa BETWEEN 9 AND 10 THEN
      v_grade := 'A+';
   ELSE
      v_grade := 'Invalid GPA';
   END IF;

   -- Display result
   DBMS_OUTPUT.PUT_LINE('RollNo: ' || v_rollno || 
                        ' | GPA: ' || v_gpa || 
                        ' | Grade: ' || v_grade);

EXCEPTION
   WHEN NO_DATA_FOUND THEN
      DBMS_OUTPUT.PUT_LINE('No student found with RollNo ' || v_rollno);
   WHEN OTHERS THEN
      DBMS_OUTPUT.PUT_LINE('Unexpected error occurred.');
END;

DECLARE
   i NUMBER := 1;
   gpa StudentTable.GPA%TYPE;
   grade VARCHAR2(2);
BEGIN
   WHILE i <= 5 LOOP

      -- Fetch GPA
      SELECT GPA INTO gpa
      FROM StudentTable
      WHERE RollNo = i;

      -- Assign grade
      IF gpa < 4 THEN
         grade := 'F';
      ELSIF gpa < 5 THEN
         grade := 'E';
      ELSIF gpa < 6 THEN
         grade := 'D';
      ELSIF gpa < 7 THEN
         grade := 'C';
      ELSIF gpa < 8 THEN
         grade := 'B';
      ELSIF gpa < 9 THEN
         grade := 'A';
      ELSE
         grade := 'A+';
      END IF;

      -- Update table
      UPDATE StudentTable
      SET LetterGrade = grade
      WHERE RollNo = i;

      -- Next student
      i := i + 1;

   END LOOP;

   DBMS_OUTPUT.PUT_LINE('Grades Updated Successfully');
END;
